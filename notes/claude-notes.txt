// 1️⃣ Inject into the real header
function injectHeaderButton(container) {
    if (container.querySelector('#promptmate-btn')) return;  // already there

    const btn = document.createElement('button');
    btn.id = 'promptmate-btn';
    btn.type = 'button';
    btn.innerText = 'PromptMate';
    btn.className = 'inline-flex items-center justify-center relative shrink-0 can-focus select-none disabled:pointer-events-none disabled:opacity-50 disabled:shadow-none disabled:drop-shadow-none text-text-300 border-transparent transition font-styrene duration-300 ease-[cubic-bezier(0.165,0.85,0.45,1)] hover:bg-bg-400 aria-pressed:bg-bg-400 aria-checked:bg-bg-400 aria-expanded:bg-bg-300 hover:text-text-100 aria-pressed:text-text-100 aria-checked:text-text-100 aria-expanded:text-text-100 h-9 px-4 py-2 rounded-lg min-w-[5rem] active:scale-[0.985] whitespace-nowrap text-sm pl-2 pr-3 gap-1 font-medium text-sm !text-text-100';
    btn.addEventListener('click', openSidebar);

    container.appendChild(btn);
}

// 2️⃣ Inject absolute button for brand-new chat (no <header> yet)
function injectNewChatButton(main) {
    if (main.querySelector('#promptmate-btn-abs')) return;  // already injected

    // ensure main can be a positioning context
    //   const prevPos = getComputedStyle(main).position;
    //   if (prevPos === 'static') {
    //     main.style.position = 'relative';
    //   }

    const btn = document.createElement('button');
    btn.id = 'promptmate-btn-abs';
    btn.type = 'button';
    btn.innerText = 'PromptMate';
    btn.setAttribute('style', 'position:absolute; top:0; right:0; margin:0.5rem; z-index:1000;');
    btn.className = 'inline-flex items-center justify-center relative shrink-0 can-focus select-none disabled:pointer-events-none disabled:opacity-50 disabled:shadow-none disabled:drop-shadow-none text-text-300 border-transparent transition font-styrene duration-300 ease-[cubic-bezier(0.165,0.85,0.45,1)] hover:bg-bg-400 aria-pressed:bg-bg-400 aria-checked:bg-bg-400 aria-expanded:bg-bg-300 hover:text-text-100 aria-pressed:text-text-100 aria-checked:text-text-100 aria-expanded:text-text-100 h-9 px-4 py-2 rounded-lg min-w-[5rem] active:scale-[0.985] whitespace-nowrap text-sm pl-2 pr-3 gap-1 font-medium text-sm !text-text-100';
    btn.addEventListener('click', openSidebar);
    main.appendChild(btn);
}

// 3️⃣ Master injector
function insertPromptMateButton() {
    const hdr = document.querySelector('header');
    if (hdr) {
        const container = hdr.querySelector('.right-3.flex.gap-2 > div');
        if (container) injectHeaderButton(container);
        return;
    }

    const main = document.querySelector('main');
    if (main) {
        injectNewChatButton(main);
    }
}

// 3️⃣ Sidebar inserter
function openSidebar() {
  // avoid opening twice
  if (document.getElementById('promptmate-sidebar')) {
    return;
  }
  // find the main flex wrapper
  const root = document.querySelector('div.flex.min-h-screen.w-full.overflow-x-hidden');
  if (!root) {
    console.warn('PromptMate: container not found');
    return;
  }
  // build your sidebar
  const sidebar = document.createElement('div');
  sidebar.id = 'promptmate-sidebar';
  sidebar.className = 'w-72 bg-white shadow-lg flex flex-col overflow-y-auto';
  // you can adjust width (w-72) and styling as needed
  
  // example: add a close button
  const closeBtn = document.createElement('button');
  closeBtn.innerText = '×';
  closeBtn.className = 'self-end m-2 text-xl';
  closeBtn.addEventListener('click', () => sidebar.remove());
  sidebar.appendChild(closeBtn);

  // (TODO) mount your React/Vanilla sidebar UI here instead of just a placeholder
  const content = document.createElement('div');
  content.innerHTML = '<h2 class="p-4 text-lg font-bold">PromptMate</h2>';
  sidebar.appendChild(content);

  // append as last child of the flex container
  root.appendChild(sidebar);
}

// 4️⃣ Kick off immediately and on SPA updates
document.addEventListener('DOMContentLoaded', insertPromptMateButton);
new MutationObserver(insertPromptMateButton)
    .observe(document.body, { childList: true, subtree: true });
